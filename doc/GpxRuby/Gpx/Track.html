def center_of_gravity
  track_points  = points
  len = track_points.size
  center = track_points.reduce(:+)
  center / len
end

def total_time
  start_time = @segments.first.points.first.time
  end_time = @segments.last.points.last.time
  end_time - start_time rescue 0
end
