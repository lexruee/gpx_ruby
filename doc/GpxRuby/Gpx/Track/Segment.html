def distance
  if @distance
    @distance
  else
    zipped_points = @points[0...-1].zip(@points[1..-1])
    distances = zipped_points.map {|(p,q)| p.distance(q) }
    @distance = distances.reduce(:+)
  end
end

def total_time
  start_time, end_time = @points.first.time, @points.last.time
  end_time - start_time rescue 0
end
